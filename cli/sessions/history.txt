+@write test.sh #!/bin/bash
+
+set -e
+
+# -------------------------------
+# Step 1: Check required tools
+# -------------------------------
+
+for tool in iw airodump-ng tcpdump; do
+    if ! command -v "$tool" &> /dev/null; then
+        echo "$tool not found. Installing..."
+        sudo apt update && sudo apt install -y aircrack-ng iw tcpdump
+        break
+    fi
+done
+
+# -------------------------------
+# Step 2: Identify Wi-Fi interface
+# -------------------------------
+
+INTERFACE=$(iw dev | awk '/Interface/ {print $2}' | head -n 1)
+
+if [ -z "$INTERFACE" ]; then
+    echo "No wireless interface found."
+    exit 1
+fi
+
+echo "Detected interface: $INTERFACE"
+
+# -------------------------------
+# Step 3: Enable monitor mode
+# -------------------------------
+
+echo "Enabling monitor mode..."
+sudo airmon-ng start "$INTERFACE" > /dev/null
+
+MON_IFACE=$(iw dev | awk '/Interface/ {print $2}' | grep mon | head -n 1)
+
+if [ -z "$MON_IFACE" ]; then
+    echo "Failed to enable monitor mode."
+    exit 1
+fi
+
+echo "Monitor interface: $MON_IFACE"
+
+# -------------------------------
+# Step 4: Start capture
+# -------------------------------
+
+echo "Starting capture..."
+sudo airodump-ng --write captured_packets "$MON_IFACE" &
+AIRODUMP_PID=$!
+
+read -n 1 -s -r -p "Press any key to stop capturing..."
+echo
+
+# -------------------------------
+# Step 5: Stop capture cleanly
+# -------------------------------
+
+echo "Stopping capture..."
+sudo kill "$AIRODUMP_PID"
+wait "$AIRODUMP_PID" 2>/dev/null
+
+# -------------------------------
+# Step 6: Disable monitor mode
+# -------------------------------
+
+echo "Disabling monitor mode..."
+sudo airmon-ng stop "$MON_IFACE" > /dev/null
+
+# -------------------------------
+# Step 7: Analyze capture (headless)
+# -------------------------------
+
+CAP_FILE="captured_packets-01.cap"
+
+if [ -f "$CAP_FILE" ]; then
+    echo "Capture file created: $CAP_FILE"
+    echo "Analyzing packets (summary):"
+    sudo tcpdump -r "$CAP_FILE"
+else
+    echo "Capture file not found."
+    exit 1
+fi
+
+echo "Done."
